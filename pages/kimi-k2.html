# MuonClip优化器核心算法
class MuonClipOptimizer:
    def clip_attention_logits(self, logits, threshold=8.0):
        """动态裁剪注意力logits，防止数值爆炸"""
        clipped = torch.clamp(logits, -threshold, threshold)
        return self.smooth_transition(clipped, logits)
    
    def adaptive_learning_rate(self, grad_norm):
        """根据梯度范数自适应调整学习率"""
        return self.base_lr / (1 + self.alpha * grad_norm)
